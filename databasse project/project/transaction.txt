CREATE TABLE C##PROJECT_USER.TRANSACTIONS (
    TRANSACTION_ID   NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    ENTITY_NAME      VARCHAR2(50), -- İşlem yapılan tablo adı
    OPERATION_TYPE   VARCHAR2(10), -- İşlem türü (INSERT, UPDATE, DELETE)
    REFERENCE_ID     NUMBER,       -- İlgili tablodaki kayıt ID'si
    OPERATION_DATE   TIMESTAMP DEFAULT SYSTIMESTAMP -- İşlem tarihi
);
